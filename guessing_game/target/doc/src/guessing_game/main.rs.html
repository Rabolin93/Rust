/*UWAGA!! caly ten program to cenna lekcja - masz dwa typy - String i str.

fn main() {
    let guess_parse_to_int :i32;
    let mut guess= String::new();

    let mut rng=rand::thread_rng();
    let random_number=rng.gen_range(0..=100);

    println!("{}",random_number);
    
    
    
    println!("Guess the number!");

    io::stdin()
        .read_line(&mut guess)
        .expect("Failed to read line"); // Tutaj korzystajac z biblioteki stdin odczytuje input i przypisuje go do
        // do guess'a, a w razie bledu wypisuje blad

    let guess_as_str=guess.as_str().trim();
    
    println!("You guess {guess}");
    guess_parse_to_int=guess_as_str.parse().expect("Something went to shit");

    println!("{guess_parse_to_int}");

    if random_number==guess_parse_to_int{
        println!("Correct answer!");
    }
    else{
        println!("Wrong answer!");
    }

    
}
